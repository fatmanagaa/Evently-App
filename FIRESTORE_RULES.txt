// Firestore Security Rules for Evently App
// Apply these rules in Firebase Console → Firestore Database → Rules tab
// Project ID: eventlyapp-1b6d7

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper functions
    function isSignedIn() {
      return request.auth.uid != null;
    }

    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    function isEventOwner(eventId) {
      return request.auth.uid == get(/databases/$(database)/documents/events/$(eventId)).data.userId;
    }

    // Events collection rules
    match /events/{eventId} {

      // Allow read if user is authenticated (for listing)
      allow read: if isSignedIn();

      // Allow create only if userId matches authenticated user
      allow create: if isSignedIn() &&
                       request.resource.data.userId == request.auth.uid &&
                       request.resource.data.title != null &&
                       request.resource.data.title is string &&
                       request.resource.data.description != null &&
                       request.resource.data.description is string &&
                       request.resource.data.category != null &&
                       request.resource.data.category is string &&
                       request.resource.data.date != null &&
                       request.resource.data.time != null &&
                       request.resource.data.createdAt != null;

      // Allow update only by event owner
      allow update: if isSignedIn() &&
                       isOwner(resource.data.userId) &&
                       request.resource.data.userId == resource.data.userId;

      // Allow delete only by event owner
      allow delete: if isSignedIn() &&
                       isOwner(resource.data.userId);

      // List events (with filters)
      allow list: if isSignedIn() &&
                     (request.query.where[0].fieldPath == ['userId'] ||
                      request.query.where[0].fieldPath == ['category']);
    }

    // Catch-all deny (security default)
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

 * 3. Navigate to Firestore Database

 *
 * RULES EXPLANATION:
 *
 * - isSignedIn(): Checks if user is authenticated with Firebase Auth
 * - isOwner(userId): Checks if current user matches the userId field
 * - isEventOwner(): Checks if current user created the event
 *
 * Events Collection:
 * - Create: Only authenticated users can create events with their own userId
 * - Read: Authenticated users can read events
 * - Update: Only event creator can update
 * - Delete: Only event creator can delete
 * - List: Supports filtering by userId or category
 *
 * Security Features:
 * - Users can only see events they created
 * - Users cannot modify other users' events
 * - Enforces required fields on creation
 * - Prevents unauthorized data modification
